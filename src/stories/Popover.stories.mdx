## Popover

## Example with default defaultPopoverItem

```jsx
import { Popover } from "@defactor/ui-kit";
import MoreHorizIcon from "@mui/icons-material/MoreHoriz";
import Box from "@mui/material/Box";
import React, { useRef, useState } from "react";

const ExamplePopoverUsage: React.FC = () => {
  const [anchorEl, setAnchorEl] = (useState < HTMLElement) | (null > null);
  const iconRef = useRef < HTMLDivElement > null;

  const handleOpen = () => {
    setAnchorEl(iconRef.current);
  };

  const handleClose = () => {
    setAnchorEl(null);
  };

  return (
    <div style={{ padding: "20px" }}>
      <Box
        ref={iconRef}
        onClick={handleOpen}
        width={150}
        textAlign={"left"}
        sx={{ cursor: "pointer" }}
      >
        <Box
          sx={{
            display: "inline-flex",
            alignItems: "center",
            justifyContent: "center",
            width: 32,
            height: 32,
            borderRadius: "50%",
            backgroundColor: "white",
            color: "#EAECF0",
            border: "1px solid #EAECF0",
          }}
        >
          <MoreHorizIcon />
        </Box>
      </Box>
      <Popover anchorEl={anchorEl} onClose={handleClose} />
    </div>
  );
};

export default ExamplePopoverUsage;
```
