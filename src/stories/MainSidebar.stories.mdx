# MainSidebar

The MainSidebar component is a navigational sidebar that includes links and icons for various sections of the application.

## Default Usage with Default Routes

```jsx
import { MainSidebar, routes } from '@defactor/ui-kit'

<MainSidebar
        hideOnBreakpoint='sm'
        notificationsCount={1}
        routes={routes.mainRoutes}
      />
```

## Default Usage with Custom Routes

```jsx
import {
  ContactsIcon,
  DashboardIcon,
  MainSidebar,
  MyTemplateIcon,
  MyWalletIcon,
  NotificationsIcon
} from '@defactor/ui-kit'

<MainSidebar
  hideOnBreakpoint='sm'
  notificationsCount={1}
  routes={[
    {
      icon: DashboardIcon,
      path: '/dashboard',
      text: 'Dashboard'
    },
    {
      icon: MyWalletIcon,
      path: '/wallet',
      text: 'MyWallet'
    },
    {
      icon: MyTemplateIcon,
      path: '/templates',
      text: 'MyTemplates'
    },
    {
      icon: NotificationsIcon,
      path: '/notifications',
      text: 'Notifications'
    },
    {
      icon: ContactsIcon,
      path: '/contacts',
      text: 'Contacts'
    }
  ]}
/>
```

```jsx
import { MainSidebar, PathProvider, routes } from '@defactor/ui-kit'
import { useRouter } from 'next/router'
import React from 'react'

const Sidebar: React.FC = () => {
  const router = useRouter() // Get the router object
  const { pathname } = router // Extract the current pathname

  return (
    <PathProvider path={pathname}>
      <MainSidebar
        hideOnBreakpoint='sm'
        notificationsCount={1}
        routes={routes.mainRoutes}
      />
    </PathProvider>
  )
}

export default Sidebar
```
