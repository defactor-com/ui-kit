## CardWithHover

`CardWithHover` is a component used to display a card with various actions such as copy, preview, and use.

### Props

<Props of={CardWithHover} />

### Example

```jsx
import React from "react";
import { CardWithHover, DocIcon } from "@defactor/ui-kit";

const Example: React.FC = () => {
  const onClickCopy = () => {
    console.log("Copy Template.");
  };

  const onClickPreview = () => {
    console.log("Preview Template.");
  };

  const onClickUse = () => {
    console.log("Use Template.");
  };

  const template = {
    id: '1',
    template_name: 'Test Card',
    description: 'This is a test card.'
  };

  return (
    <CardWithHover
      template={template}
      isPublished={true}
      onClickCopy={onClickCopy}
      onClickPreview={onClickPreview}
      onClickUse={onClickUse}
    />
  );
};

export default Example;
```
