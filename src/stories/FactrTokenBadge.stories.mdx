## FactrTokenBadge

`FactrTokenBadge` is a component that displays a token's name, percentage, and an optional icon, wrapped inside a styled badge with a percentage circle.

### Props

<Props of={FactrTokenBadge} />

### Example 1

This example shows the basic usage of the `FactrTokenBadge` with an Ethereum icon and a 68.41% percentage.

```jsx
import React from "react";
import { Box } from '@mui/material';
import FactrTokenBadge from "@defactor/ui-kit";
import { EthFactrIcon } from "@defactor/ui-kit";

const Example1: React.FC = () => {
    return (
        <Box display="flex" width="100%" alignItems="center" justifyContent="center" padding={6} sx={{ backgroundColor: "white" }}>
            <FactrTokenBadge
                icon={<EthFactrIcon />}
                name="Ethereum"
                percentage="68.41%"
            />
        </Box>
    );
};

export default Example1;
